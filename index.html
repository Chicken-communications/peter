// ============================================================ // MULTIPLAYER WEBSOCKET CONNECTION // ============================================================ let ws = null; let playerId = null; let myPlayerIndex = null; function connectToServer() { const host = window.location.hostname; const port = window.location.port || '3000'; const wsUrl = `ws://${host}:${port}`; console.log(`🔗 Connecting to ${wsUrl}`); ws = new WebSocket(wsUrl); ws.onopen = () => { console.log('✅ Connected to server'); const playerName = document.getElementById('pname-0')?.value || 'Player 1'; ws.send(JSON.stringify({ type: 'join', roomId: 'default', playerName })); }; ws.onmessage = (event) => { const msg = JSON.parse(event.data); console.log('📨 Server message:', msg.type); if (msg.type === 'init') { playerId = msg.playerId; console.log('Player ID:', playerId); } if (msg.type === 'playerJoined') { console.log('Players in game:', msg.players); setEvent('🎮 Multiplayer', msg.players.length + ' players connected!'); } if (msg.type === 'gameStateSync') { // Another player's game state — update your board view syncRemoteGameState(msg.data); } if (msg.type === 'playerLeft') { console.log('Players remaining:', msg.players); setEvent('⚠️ Player Disconnected', msg.players.length + ' players left'); } }; ws.onerror = (err) => console.error('❌ WebSocket error:', err); ws.onclose = () => console.log('🔌 Disconnected from server'); } function syncGameState() { if (ws && ws.readyState === WebSocket.OPEN) { console.log('📤 Syncing game state...'); ws.send(JSON.stringify({ type: 'gameStateUpdate', data: G })); } } function syncRemoteGameState(remoteG) { // Update board with remote player state console.log('🔄 Updating from remote state'); if (remoteG.players) { G.players = remoteG.players; G.cur = remoteG.cur; G.month = remoteG.month; G.jackpot = remoteG.jackpot; G.log = remoteG.log; renderPlayers(); placePieces(); updateInfo(); } } // ============================================================
